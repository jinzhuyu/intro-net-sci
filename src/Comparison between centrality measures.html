def draw(G, pos, measure, measure_name):
    plt.figure(figsize =(7, 4.75))
    nodes = nx.draw_networkx_nodes(G, pos, node_size=270, cmap=plt.cm.rainbow, 
                                   node_color=list(measure.values()),
                                   nodelist=measure.keys())
    nodes.set_norm(mcolors.SymLogNorm(linthresh=0.01, linscale=1, base=10))
    
    labels = nx.draw_networkx_labels(G, pos)
    edges = nx.draw_networkx_edges(G, pos)

    plt.title(measure_name, fontsize=16)
    plt.colorbar(nodes)
    plt.axis('off')
    plt.show()

def rank(measure, measure_name, is_sort=True):
    n_node = len(measure)
    if is_sort:
        measure = dict( sorted(measure.items(), key=lambda x: x[1], reverse=True) )
        color = plt.cm.rainbow(np.linspace(1, 0, n_node))
    else:
        color = 'tab:blue'
    plt.figure(figsize =(11, 3))
    plt.bar(range(n_node), list(measure.values()), align='center', color=color)
    plt.xticks(range(n_node), list(measure.keys()))
    plt.ylabel(measure_name, fontsize=16)
    plt.show()
